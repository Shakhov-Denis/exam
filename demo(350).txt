import tkinter as tk
from tkinter import ttk, messagebox
import sqlite3

# Создание подключения к базе данных SQLite
conn = sqlite3.connect("exam_system.db")
conn.row_factory = sqlite3.Row
cursor = conn.cursor()
cursor.execute("PRAGMA foreign_keys = ON")

# Создание таблиц базы данных
cursor.execute("""
CREATE TABLE IF NOT EXISTS roles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE
)
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    login TEXT NOT NULL UNIQUE,
    password TEXT NOT NULL,
    role_id INTEGER NOT NULL,
    FOREIGN KEY (role_id) REFERENCES roles(id)
)
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS categories (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE
)
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS products (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    category_id INTEGER NOT NULL,
    price REAL NOT NULL,
    quantity INTEGER NOT NULL,
    description TEXT,
    FOREIGN KEY (category_id) REFERENCES categories(id)
)
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS clients (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    full_name TEXT NOT NULL,
    phone TEXT NOT NULL,
    email TEXT,
    address TEXT
)
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS orders (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    client_id INTEGER NOT NULL,
    user_id INTEGER NOT NULL,
    order_date TEXT DEFAULT CURRENT_DATE,
    status TEXT NOT NULL DEFAULT 'Новый',
    FOREIGN KEY (client_id) REFERENCES clients(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
)
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS order_items (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_id INTEGER NOT NULL,
    product_id INTEGER NOT NULL,
    quantity INTEGER NOT NULL,
    price REAL NOT NULL,
    FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id)
)
""")



conn.commit()

# Добавление ролей в таблицу roles
for role in ["Администратор", "Менеджер", "Пользователь"]:
    cursor.execute("INSERT OR IGNORE INTO roles(name) VALUES (?)", (role,))

# Добавление категорий в таблицу categories (если они еще не добавлены)
for category in ["Техника", "Канцелярия", "Одежда", "Продукты"]:
    cursor.execute("INSERT OR IGNORE INTO categories(name) VALUES (?)", (category,))

default_users = [
    ("admin", "1234", "Администратор"),
    ("manager", "1234", "Менеджер"),
    ("user", "1234", "Пользователь")
]
for login, pwd, role_name in default_users:
    cursor.execute("SELECT id FROM roles WHERE name = ?", (role_name,))
    role_id = cursor.fetchone()[0]
    cursor.execute("INSERT OR IGNORE INTO users (login, password, role_id) VALUES (?, ?, ?)",
                   (login, pwd, role_id))

conn.commit()

def login():
    login_text = entry_login.get()
    password_text = entry_password.get()

    cursor.execute("""
    SELECT users.id, users.login, roles.name AS role
    FROM users
    JOIN roles ON users.role_id = roles.id
    WHERE users.login=? AND users.password=?
    """, (login_text, password_text))

    user = cursor.fetchone()

    if user:
        login_window.destroy()
        open_main_window(user["id"], user["login"], user["role"])
    else:
        messagebox.showerror("Ошибка", "Неверный логин или пароль")

# Главное окно программы
def open_main_window(user_id, login_name, role):
    root = tk.Tk()
    root.title("Информационная система")
    root.geometry("1200x760")

    tk.Label(root, text=f"Пользователь: {login_name} | Роль: {role}", font=("Times new roman", 13, "bold")).pack(pady=10)

    notebook = ttk.Notebook(root)
    notebook.pack(fill="both", expand=True)

    products_tab = tk.Frame(notebook)
    clients_tab = tk.Frame(notebook)
    orders_tab = tk.Frame(notebook)

    notebook.add(products_tab, text="Товары")
    notebook.add(clients_tab, text="Клиенты")
    notebook.add(orders_tab, text="Заказы")

    selected_product_id = tk.IntVar(value=0)

    # --- Формы для добавления товара ---
    def add_product():
        name = product_name.get()
        category = product_category.get()
        price = product_price.get()
        quantity = product_quantity.get()
        description = product_description.get()

        if not name or not category or not price or not quantity:
            messagebox.showerror("Ошибка", "Заполните все поля")
            return
        category_id = get_category_id(category)
        if not category_id:
            messagebox.showerror("Ошибка", "Категория товара не найдена")
            return

        cursor.execute("""
        INSERT INTO products (name, category_id, price, quantity, description)
        VALUES (?, ?, ?, ?, ?)
        """, (name, category_id, float(price), int(quantity), description))

        conn.commit()
        load_products()

    def load_products():
        for row in product_table.get_children():
            product_table.delete(row)

        cursor.execute("""
        SELECT products.id, products.name, categories.name AS category, products.price, products.quantity, products.description
        FROM products
        JOIN categories ON products.category_id = categories.id
        """)
        for row in cursor.fetchall():
            product_table.insert("", tk.END, values=(
                row["id"], row["name"], row["category"], row["price"], row["quantity"], row["description"]
            ))

    def get_category_id(category_name):
        cursor.execute("SELECT id FROM categories WHERE name=?", (category_name,))
        row = cursor.fetchone()
        return row["id"] if row else None

    # Поля ввода для товара
    tk.Label(products_tab, text="Название товара:").pack()
    product_name = tk.Entry(products_tab)
    product_name.pack()

    tk.Label(products_tab, text="Категория товара:").pack()
    product_category = ttk.Combobox(products_tab)
    product_category.pack()

    # Заполнение категорий в выпадающем списке
    cursor.execute("SELECT name FROM categories")
    categories = [row["name"] for row in cursor.fetchall()]
    product_category['values'] = categories

    tk.Label(products_tab, text="Цена товара:").pack()
    product_price = tk.Entry(products_tab)
    product_price.pack()

    tk.Label(products_tab, text="Количество товара:").pack()
    product_quantity = tk.Entry(products_tab)
    product_quantity.pack()

    tk.Label(products_tab, text="Описание товара:").pack()
    product_description = tk.Entry(products_tab)
    product_description.pack()

    tk.Button(products_tab, text="Добавить товар", command=add_product).pack()

    # Таблица товаров
    product_columns = ("ID", "Название", "Категория", "Цена", "Количество", "Описание")
    product_table = ttk.Treeview(products_tab, columns=product_columns, show="headings")
    for col in product_columns:
        product_table.heading(col, text=col)
        product_table.column(col, width=170)
    product_table.pack(fill="both", expand=True, pady=10)

    # Загрузка товаров
    load_products()

    # --- Формы для клиентов ---
    def add_client():
        name = client_name.get()
        phone = client_phone.get()
        email = client_email.get()
        address = client_address.get()

        if not name or not phone or not address:
            messagebox.showerror("Ошибка", "Заполните все поля")
            return

        cursor.execute("""
        INSERT INTO clients (full_name, phone, email, address)
        VALUES (?, ?, ?, ?)
        """, (name, phone, email, address))

        conn.commit()
        load_clients()

    def load_clients():
        for row in client_table.get_children():
            client_table.delete(row)

        cursor.execute("""
        SELECT id, full_name, phone, email, address
        FROM clients
        """)
        for row in cursor.fetchall():
            client_table.insert("", tk.END, values=(
                row["id"], row["full_name"], row["phone"], row["email"], row["address"]
            ))

    # Поля ввода для клиента
    tk.Label(clients_tab, text="Имя клиента:").pack()
    client_name = tk.Entry(clients_tab)
    client_name.pack()

    tk.Label(clients_tab, text="Телефон клиента:").pack()
    client_phone = tk.Entry(clients_tab)
    client_phone.pack()

    tk.Label(clients_tab, text="Email клиента:").pack()
    client_email = tk.Entry(clients_tab)
    client_email.pack()

    tk.Label(clients_tab, text="Адрес клиента:").pack()
    client_address = tk.Entry(clients_tab)
    client_address.pack()
    
    tk.Button(clients_tab, text="Добавить клиента", command=add_client).pack()

    # Таблица клиентов
    client_columns = ("ID", "Имя", "Телефон", "Email", "Адрес")
    client_table = ttk.Treeview(clients_tab, columns=client_columns, show="headings")
    for col in client_columns:
        client_table.heading(col, text=col)
        client_table.column(col, width=170)
    client_table.pack(fill="both", expand=True, pady=10)

    # Загрузка клиентов
    load_clients()

    # --- Формы для заказов ---
    def add_order():
        client_id = client_id_order.get()
        user_id = user_id_order.get()

        if not client_id or not user_id:
            messagebox.showerror("Ошибка", "Заполните все поля")
            return

        cursor.execute("""
        INSERT INTO orders (client_id, user_id)
        VALUES (?, ?)
        """, (client_id, user_id))

        conn.commit()
        load_orders()

    def load_orders():
        for row in order_table.get_children():
            order_table.delete(row)

        cursor.execute("""
        SELECT orders.id, clients.full_name, users.login, orders.order_date, orders.status
        FROM orders
        JOIN clients ON orders.client_id = clients.id
        JOIN users ON orders.user_id = users.id
        """)
        for row in cursor.fetchall():
            order_table.insert("", tk.END, values=(
                row["id"], row["full_name"], row["login"], row["order_date"], row["status"]
            ))

    # Поля ввода для заказа
    tk.Label(orders_tab, text="ID клиента:").pack()
    client_id_order = tk.Entry(orders_tab)
    client_id_order.pack()

    tk.Label(orders_tab, text="ID пользователя:").pack()
    user_id_order = tk.Entry(orders_tab)
    user_id_order.pack()

    tk.Button(orders_tab, text="Добавить заказ", command=add_order).pack()

    # Таблица заказов
    order_columns = ("ID", "Клиент", "Пользователь", "Дата заказа", "Статус")
    order_table = ttk.Treeview(orders_tab, columns=order_columns, show="headings")
    for col in order_columns:
        order_table.heading(col, text=col)
        order_table.column(col, width=170)
    order_table.pack(fill="both", expand=True, pady=10)

    # Загрузка заказов
    load_orders()

    # --- Запуск программы ---
    root.mainloop()
